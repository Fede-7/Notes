int var1 = 50;
int main()
{
    pid_t pid1, pid2;
    int var2 = 100;
    pid2 = fork();
    if(pid1 == 0){
        var1 = var1 + var2;
        pid2 = fork();
    } else {
        pid2 = fork();
        var2 = var2 + pid1 + pid2;
    }
    print(f"pid=%d var1=%d \n", getpid(), var1, var2);
    return 0;
}
